import matplotlib.pyplot as plt

def plot_result(statNms,legendArr, legendNm,data_type, ylabel, number=0):
    
    if ylabel == 'Cross Entropy':
        tail = '_ce'
    elif ylabel == 'Accuracy':
        tail = '_acc'
    else:
        tail = '_' + ylabel
   
    #plt.figure(figsize=(6.40, 6.80), dpi=100)
    
    plt.figure(number)
    plt.clf()
    cnt = 0
    for statNm in statNms:
        dataArr = np.array(stats[statNm][data_type + tail])
        plt.plot(dataArr[:, 0], dataArr[:, 1], label=legendNm + " " +legendArr[cnt])
        cnt+=1
    plt.xlabel('Epoch')
    plt.ylabel(data_type+ " " +ylabel)
    plt.legend()
    plt.draw()
